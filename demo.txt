🧪 PRACTICAL 4C — PARAMETERIZED JOB
✅ Freestyle
Enable:
This project is parameterized

Parameter:
Name = NAME
Default = Student
Build → Execute Shell
echo "Hello $NAME"
echo "Build triggered by $NAME"
date
✅ Parameterized Pipeline
pipeline {
    agent any

    parameters {
        string(name: 'NAME', defaultValue: 'Student', description: 'Enter name')
    }

    triggers {
        cron('H/2 * * * *')
    }

    stages {
        stage('Greet') {
            steps {
                echo "Hello ${params.NAME}"
            }
        }

        stage('File Creation') {
            steps {
                sh '''
                echo "Hello ${NAME}" > message.txt
                '''
            }
        }
    }

    post {
        success {
            archiveArtifacts artifacts: 'message.txt'
        }
    }
}
